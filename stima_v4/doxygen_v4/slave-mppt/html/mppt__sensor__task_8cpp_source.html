70 {

75  {
76  // Check 1/2 Freq. controller ready to WDT only SET flag

151  break;
152

155  // Start with measure ready also with VIN < VBAT

167  is_power_down = true;

170

172  edata.value = param.mpptIC->get_P_CHG(&is_measure_done);
173  is_error_measure |= !is_measure_done;
174  // Power % > 70% (Full OK)
175  // Power % > 30% (No Critical)
176  if(edata.value > BATTERY_CHARGE_MIN_FULL) is_power_full = true;
177  if(edata.value > BATTERY_CHARGE_MIN_WARNING) is_power_warning = false;
178  if(edata.value > BATTERY_CHARGE_MIN_CRITICAL) is_power_critical = false;
179  if(edata.value > BATTERY_CHARGE_MIN_EMPTY) is_power_down = false;
180  edata.index = POWER_BATTERY_CHARGE_INDEX;
181  param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
182

184  is_error_measure |= !is_measure_done;
185  edata.index = POWER_BATTERY_CURRENT_INDEX;
186  param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
187 
188  edata.value = param.mpptIC->get_V_IN(&is_measure_done) * POWER_INPUT_VOLTAGE_MULT;
189  is_error_measure |= !is_measure_done;
190  // VIn Without VBAT (Full OK). Direct powered from external power supply.
191  if((edata.value > INPUT_VOLTAGE_MIN_POWERDOWN) && (is_power_down)) {
192  is_power_full = true;
193  is_power_warning = false;
194  is_power_critical = false;
195  }
196  edata.index = POWER_INPUT_VOLTAGE_INDEX;
197  param.elaborateDataQueue->Enqueue(&edata, Ticks::MsToTicks(WAIT_QUEUE_REQUEST_PUSHDATA_MS));
198 
199  edata.value = param.mpptIC->get_I_IN(&is_measure_done) * POWER_INPUT_CURRENT_MULT;
200  is_error_measure |= !is_measure_done;
201  // IIn > 250 mA (Full OK)
202  if(edata.value > 250) {
203  is_power_full = true;
204  is_power_warning = false;
205  is_power_critical = false;
206  }
207 
208  // Response system_status power level event flags
209  param.systemStatusLock->Take();
210  param.system_status->events.is_ltc_unit_error = is_error_measure;
211  param.system_status->events.is_power_full = is_power_full;
212  param.system_status->events.is_power_warning = is_power_warning;
213  param.system_status->events.is_power_critical = is_power_critical;
214  param.systemStatusLock->Give();
215 
216  state = SENSOR_STATE_END;
217  break;
218 
219  case SENSOR_STATE_END:
220 
221  #if (ENABLE_STACK_USAGE)
222  TaskMonitorStack();
223  #endif
224 
225  // Local TaskWatchDog update and Sleep Activate before Next Read

227  TaskState(state, UNUSED_SUB_POSITION, task_flag::sleepy);
228  DelayUntil(Ticks::MsToTicks(param.configuration->sensor_acquisition_delay_ms));
229  TaskState(state, UNUSED_SUB_POSITION, task_flag::normal);
230 
231  state = SENSOR_STATE_READ;
232  break;
233  }
