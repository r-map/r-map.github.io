213 bool Mppt::smbus_read_reg(uint16_t reg, uint16_t *data)
214 {
215  // Try lock Semaphore
216  if (_wireLock->Take(Ticks::MsToTicks(1000)))
217  {
218  uint8_t lsb;
219  uint8_t msb;
220  /* Read SMBus register LTC4015 */
221  _wire->beginTransmission(_i2c_address);
222  _wire->write(reg);
223  _wire->endTransmission();
224  // Read Operation request x Word Register
225  _wire->requestFrom(_i2c_address, 2);
226  lsb = _wire->read();
227  msb = _wire->read();
228  _wire->endTransmission();
229  _wireLock->Give();
230  *data = (msb << 8) | lsb;
231  return true;
232  } else
233  return false;
234 }
