227 uint8_t ElaborateDataTask::checkWindQuality(float speed, float direction) {
228  // Optional check quality data function
229  float check_speed;
230  float check_direction;
231  uint8_t quality = 0;
232 
233  // Spike on Velocity > 10 m/sec on near acquire (1 samp / 1 sec.)
234  check_speed = fabs(speed - speed_last);
235  if(check_speed < 10) quality += 50;
236  else quality += check_speed - 50;
237  // Check difference (90°) between 2 near acquire (1 samp / 1 sec.)
238  check_direction = fabs(direction - direction_last);
239  if((check_direction > 180.0)) check_direction = 360.0 - check_direction;
240  if(check_direction < 90) quality += 50;
241  else quality += (0.55556 * (check_direction - 90.0));
242 
243  // Limit control
244  if(quality < 0) quality = 0;
245  if(quality > 100) quality = 100;
246 
247  // Backup value
248  speed_last = speed;
249  direction_last = direction;
250 
251  return quality;
252 }

259 void ElaborateDataTask::getSDFromUV(float u, float v, float *speed, float *direction)
260 {
261  // Check limit for ATAN2 function near 0. Undefined result on 0
262  if ((abs(u) < ATAN2_CHECK_LIMIT) && (abs(v) < ATAN2_CHECK_LIMIT)) {
263  *speed = 0;
264  *direction = 0;
265  } else {
266  *speed = sqrt(u * u + v * v);
267  *direction = RAD_TO_DEG * atan2(-u, -v);
268  int16_t tmp_dir = round(*direction);
269  if(tmp_dir < 0) tmp_dir += 360;
270  *direction = tmp_dir % 360;
271  if(*speed < CALM_WIND_MAX_MED_VECT) {
272  *speed = 0;
273  *direction = 0;
274  } else if (*direction == 0) {
275  *direction = WIND_DIRECTION_MAX; // Translate 0 -> 360 (WIND_DIRECTION_MAX)
276  }
277  }
