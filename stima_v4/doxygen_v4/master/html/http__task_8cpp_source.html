63 void HttpTask::TaskWatchDog(uint32_t millis_standby)
64 {
65  // Local TaskWatchDog update
66  param.systemStatusLock->Take();
67  // Update WDT Signal (Direct or Long function Timered)
68  if(millis_standby)
69  {
70  // Check 1/2 Freq. controller ready to WDT only SET flag
71  if((millis_standby) < WDT_CONTROLLER_MS / 2) {
72  param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
73  } else {
74  param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::timer;
75  // Add security milimal Freq to check
76  param.system_status->tasks[LOCAL_TASK_ID].watch_dog_ms = millis_standby + WDT_CONTROLLER_MS;
77  }
78  }
79  else
80  param.system_status->tasks[LOCAL_TASK_ID].watch_dog = wdt_flag::set;
81  param.systemStatusLock->Give();
82 }

562  error = ERROR_END_OF_STREAM;
563  TRACE_ERROR_F(F("%s Firmware module not avaiable on server RMAP, next checking...\r\n"), Thread::GetName().c_str());
564  }
565  else if (bErrorFirmwareDownload)
566  {
567  // Closing queue and removing file
568  do_firmware_end_data(false);
569  // Error if of queue/SD no more action here, exit and signal the problem to server (SD)
570  is_error = true;
571  state = HTTP_STATE_END;
572  TRACE_ERROR_F(F("%s Failed to upload firmware stream to SD [ %s ]\r\n"), Thread::GetName().c_str(), ABORT_STRING);
573  break;
574  }
575  }
576 
577  // Any error to report?
578  if (error != ERROR_END_OF_STREAM)

581  if ((is_get_firmware)&&(bValidFirmwareRequest)) {
582  do_firmware_end_data(false);
583  }

587  state = HTTP_STATE_END;
588  TRACE_ERROR_F(F("%s Failed to parse http stream [ %s ] ABORT!!!\r\n"), Thread::GetName().c_str(), ERROR_STRING);
589  } else {
590  TaskWatchDog(HTTP_TASK_GENERIC_RETRY_DELAY_MS);
591  Delay(Ticks::MsToTicks(HTTP_TASK_GENERIC_RETRY_DELAY_MS));
592  TRACE_ERROR_F(F("%s Failed to parse http stream [ %s ]\r\n"), Thread::GetName().c_str(), ERROR_STRING);
593  }
594  break;
595  }
596 
597  // Closing Queue and File data if opened (Ready for next firmware...)
598  if ((is_get_firmware)&&(bValidFirmwareRequest)) {
599  // Send closing File correct complete Data Chunk
600  bErrorFirmwareDownload |= do_firmware_end_data(true);
601  // At least one firmware are downloaded. Need to resynch version/revision to check avaiables version into SD card...
602  bDownloadedFirmware = true;
603  }
604 
605  // Close HTTP response body
606  error = httpClientCloseBody(&httpClientContext);
607  // Any error to report?
608  if (error)
609  {
610  if(++retry_get_response<HTTP_TASK_GENERIC_RETRY) {
611  is_error = true;
612  state = HTTP_STATE_END;
613  TRACE_ERROR_F(F("%s Failed to read http response trailer [ %s ] ABORT!!!\r\n"), Thread::GetName().c_str(), ERROR_STRING);
614  } else {
615  TaskWatchDog(HTTP_TASK_GENERIC_RETRY_DELAY_MS);
616  Delay(Ticks::MsToTicks(HTTP_TASK_GENERIC_RETRY_DELAY_MS));
617  TRACE_ERROR_F(F("%s Failed to read http response trailer [ %s ]\r\n"), Thread::GetName().c_str(), ERROR_STRING);
618  }
619  break;
620  }
621 
622  // Gracefully disconnect from the HTTP server
623  httpClientDisconnect(&httpClientContext);
624

626  if(is_get_firmware)
627  {
628  state = HTTP_STATE_LOOP_REQUEST_FIRMWARE;
629  TRACE_VERBOSE_F(F("HTTP_STATE_GET_RESPONSE -> HTTP_STATE_LOOP_REQUEST_FIRMWARE\r\n"));
630  }
631  else
632  {
633  state = HTTP_STATE_END;
634  TRACE_VERBOSE_F(F("HTTP_STATE_GET_RESPONSE -> HTTP_STATE_END\r\n"));
635  }
636  break;
637 
638  // Loop for module enabled and configure to check download firmware
639  case HTTP_STATE_LOOP_REQUEST_FIRMWARE:
640  while(true) {
641  // ( From Master to first slave FF -> 00 start to first module )
642  // Ending Module? End Of Procedure
643  if(++module_download >= BOARDS_COUNT_MAX) break;
644  if(param.system_status->data_slave[module_download].module_version) {
645  TRACE_INFO_F(F("Http: Starting next firmware request download module: [ %d ] \r\n"), param.system_status->data_slave[module_download].module_type);

647  // Firmware download request can be start normally
648  break;
649  }
650  }
651  if(module_download >= BOARDS_COUNT_MAX) {
652  // End of loop downloading firmware
653  is_error = NO_ERROR;
654  state = HTTP_STATE_END;
655  TRACE_VERBOSE_F(F("HTTP_STATE_LOOP_REQUEST_FIRMWARE -> HTTP_STATE_END\r\n"));
656 
657  // Resynch firmware SD with avaiables ONLY!!! if new version are downloaded from server
658  if (bDownloadedFirmware) {
659  // After all download firmware module start queue request reload structure firmware
660  // And waiting response. After start update all firmware board status info are resynch
661  system_message_t system_message = {0};
662  system_message.task_dest = SD_TASK_ID;
663  system_message.command.do_reload_fw = true;
664  system_message.param = CMD_PARAM_REQUIRE_RESPONSE;
665  param.systemMessageQueue->Enqueue(&system_message);
666 
667  // Waiting a response done before continue (reload status flag firmware OK!!!)
668  while(true) {
669  // Continuos Switching context non blocking
670  // Need Waiting Task for start command on All used TASK
671  taskYIELD();
672  vTaskDelay(100);
673  // Check response done
674  if(!param.systemMessageQueue->IsEmpty()) {
675  param.systemMessageQueue->Peek(&system_message);
676  if(system_message.command.done_reload_fw) {
677  // Remove message (Reload Done is OK)
678  param.systemMessageQueue->Dequeue(&system_message);
679  break;
680  }
681  }
682  }
683  // Reset version and revision (force reload difference after download firmware to verify)
684  // If new version found, flag new_firmware_ready must be set to true on new connection...
685  param.systemStatusLock->Take();
686  for (uint8_t i = 0; i < BOARDS_COUNT_MAX; i++) {
687  param.system_status->data_slave[i].module_version = 0;
688  param.system_status->data_slave[i].module_revision = 0;
689  }
690  param.systemStatusLock->Give();
691  }
692  }
693  else
694  {
695  // Deinit Current context and restart next
696  httpClientDeinit(&httpClientContext);
697  is_error = NO_ERROR;
698  state = HTTP_STATE_SEND_REQUEST;
699  TRACE_VERBOSE_F(F("HTTP_STATE_LOOP_REQUEST_FIRMWARE -> HTTP_STATE_SEND_REQUEST\r\n"));
700  }
701  break;
702 
703  case HTTP_STATE_END:
704  // ok
705  if (!is_error)
706  {
707  param.systemStatusLock->Take();
708  param.system_status->flags.http_error = false;
709  param.system_status->connection.is_http_configuration_updating = false;
710  param.system_status->connection.is_http_configuration_updated = is_get_configuration;
711  param.system_status->connection.is_http_firmware_upgrading = false;
712  param.system_status->connection.is_http_firmware_upgraded = is_get_firmware;
713  param.systemStatusLock->Give();
714 
715  httpClientDeinit(&httpClientContext);
716 
717  memset(&connection_response, 0, sizeof(connection_response_t));
718  connection_response.done_http_configuration_getted = is_get_configuration;
719  connection_response.error_http_configuration_getted = false;
720  connection_response.done_http_firmware_getted = is_get_firmware;
721  connection_response.error_http_firmware_getted = false;
722  param.connectionResponseQueue->Enqueue(&connection_response);
723 
724  state = HTTP_STATE_INIT;
725  TRACE_VERBOSE_F(F("HTTP_STATE_END -> HTTP_STATE_INIT\r\n"));
726  }
727  // retry (DNS Error is connection error than required forced reset connection)
728  else if (((++retry) < HTTP_TASK_GENERIC_RETRY) && (!param.system_status->connection.is_dns_failed_resolve))
729  {
730  TaskWatchDog(HTTP_TASK_GENERIC_RETRY_DELAY_MS);
731  Delay(Ticks::MsToTicks(HTTP_TASK_GENERIC_RETRY_DELAY_MS));
732 
733  TRACE_VERBOSE_F(F("HTTP_STATE_END -> HTTP_STATE_SEND_REQUEST\r\n"));
734  state = HTTP_STATE_SEND_REQUEST;
735  }
736  // error
737  else
738  {
739  param.systemStatusLock->Take();
740  param.system_status->flags.http_error = true;
741  param.system_status->connection.is_http_configuration_updating = false;
742  param.system_status->connection.is_http_configuration_updated = false;
743  param.system_status->connection.is_http_firmware_upgrading = false;
744  param.system_status->connection.is_http_firmware_upgraded = false;
745  param.systemStatusLock->Give();
746 
747  httpClientDeinit(&httpClientContext);
748 
749  memset(&connection_response, 0, sizeof(connection_response_t));
750  connection_response.done_http_configuration_getted = false;
751  connection_response.error_http_configuration_getted = is_get_configuration;
752  connection_response.done_http_firmware_getted = false;
753  connection_response.error_http_firmware_getted = is_get_firmware;
754  param.connectionResponseQueue->Enqueue(&connection_response);
755 
756  state = HTTP_STATE_INIT;
757  TRACE_VERBOSE_F(F("HTTP_STATE_END -> HTTP_STATE_INIT\r\n"));
758  }
759  break;
760 
761  // One step base non blocking switch
762  TaskWatchDog(HTTP_TASK_WAIT_DELAY_MS);
763  Delay(Ticks::MsToTicks(HTTP_TASK_WAIT_DELAY_MS));
764 
765  }
766  }
