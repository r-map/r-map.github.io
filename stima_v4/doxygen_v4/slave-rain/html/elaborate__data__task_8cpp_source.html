282 uint8_t ElaborateDataTask::checkRain(void) {
283  float quality = 100.0;
284 
285  #if (USE_CLOGGED_UP_CONTROL)
286  // Checking signal CLOGGED_UP (on Event or Reset... remote calling)
287  param.systemStatusLock->Take();
288  #if (CLOGGED_EVENT_VALUE)
289  param.system_status->events.is_clogged_up = digitalRead(CLOGGED_UP_PIN);
290  #else
291  param.system_status->events.is_clogged_up = !digitalRead(CLOGGED_UP_PIN);
292  #endif
293  param.systemStatusLock->Give();
294  if(param.system_status->events.is_clogged_up) {
295  TRACE_INFO_F(F("Sensor: read status of clogged up... [ ALERT ]\r\n"));
296  }
297  #endif
298 
299  if(param.system_status->events.is_clogged_up) {
300  quality = 0.0;
301  } else {
302  if(param.system_status->events.is_tipping_error) {
303  quality -= param.system_status->events.error_count; // 1 Error = -1%
304  if(quality<0.0) quality = 0.0;
305  }
306  // Reduce 15% for error on one reed
307  if(param.system_status->events.is_main_error) quality *= 0.85;
308  if(param.system_status->events.is_redundant_error) quality *= 0.85;
309  // Reduce 30% on error bubble_level (if accelerometr working correctly)
310  if((!param.system_status->events.is_accelerometer_error) &&
311  (param.system_status->events.is_bubble_level_error)) quality *= 0.7;
312  }
313  return (uint8_t) quality;
