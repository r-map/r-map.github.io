1 /*
2  * kk_ihex_read.h: A simple library for reading Intel HEX data. See
3  * the accompanying kk_ihex_write.h for IHEX write support.
4  *
5  *
6  * READING INTEL HEX DATA
7  * ----------------------
8  *
9  * To read data in the Intel HEX format, you must perform the actual reading
10  * of bytes using other means (e.g., stdio). The bytes read must then be
11  * passed to `ihex_read_byte` and/or `ihex_read_bytes`. The reading functions
12  * will then call `ihex_data_read`, at which stage the `struct ihex_state`
13  * structure will contain the data along with its address. See below for
14  * details and example implementation of `ihex_data_read`.
15  *
16  * The sequence to read data in IHEX format is:
17  * struct ihex_state ihex;
18  * ihex_begin_read(&ihex);
19  * ihex_read_bytes(&ihex, my_input_bytes, length_of_my_input_bytes);
20  * ihex_end_read(&ihex);
21  *
22  *
23  * CONSERVING MEMORY
24  * -----------------
25  *
26  * For memory-critical use, you can save additional memory by defining
27  * `IHEX_LINE_MAX_LENGTH` as something less than 255. Note, however, that
28  * this limit affects both reading and writing, so the resulting library
29  * will be unable to read lines with more than this number of data bytes.
30  * That said, I haven't encountered any IHEX files with more than 32
31  * data bytes per line.
32  *
33  *
34  * Copyright (c) 2013-2019 Kimmo Kulovesi, https://arkku.com/
35  * Provided with absolutely no warranty, use at your own risk only.
36  * Use and distribute freely, mark modified copies as such.
37  */

71 // Called when a complete line has been read, the record type of which is
72 // passed as `type`. The `ihex` structure will have its fields `data`,
73 // `line_length`, `address`, and `segment` set appropriately. In case
74 // of reading an `IHEX_EXTENDED_LINEAR_ADDRESS_RECORD` or an
75 // `IHEX_EXTENDED_SEGMENT_ADDRESS_RECORD` the record's data is not
76 // yet parsed - it will be parsed into the `address` or `segment` field
77 // only if `ihex_data_read` returns `true`. This allows manual handling
78 // of extended addresses by parsing the `ihex->data` bytes.
79 //
80 // Possible error cases include checksum mismatch (which is indicated
81 // as an argument), and excessive line length (in case this has been
82 // compiled with `IHEX_LINE_MAX_LENGTH` less than 255) which is indicated
83 // by `line_length` greater than `length`. Unknown record types and
84 // other erroneous data is usually silently ignored by this minimalistic
85 // parser. (It is recommended to compute a hash over the complete data
86 // once received and verify that against the source.)
87 //
88 // Example implementation:
89 //
90 // ihex_bool_t ihex_data_read(struct ihex_state *ihex,
91 // ihex_record_type_t type,
92 // ihex_bool_t error) {
93 // error = error || (ihex->length < ihex->line_length);
94 // if (type == IHEX_DATA_RECORD && !error) {
95 // (void) fseek(outfile, IHEX_LINEAR_ADDRESS(ihex), SEEK_SET);
96 // (void) fwrite(ihex->data, 1, ihex->length, outfile);
97 // } else if (type == IHEX_END_OF_FILE_RECORD) {
98 // (void) fclose(outfile);
99 // }
100 // return !error;
101 // }
102 //
103 extern ihex_bool_t ihex_data_read(struct ihex_state *ihex,
104  ihex_record_type_t type,
105  ihex_bool_t checksum_mismatch);
